--@include /lib/nablib.lua.txt
require( "/lib/nablib.lua.txt" )

NAB = {
    root = "nabtech/",
    bin = "nabtech/bin/",
    cfg = "nabtech/cfg/",
    data = "nabtech/data/",
}

local initialized = false
local init
if CLIENT then
	init = function()
		if player() ~= owner() then return end
		if initialized then return end
		initialized = true
		
		include( NAB.bin .. "client.lua.txt" )
	end
elseif SERVER then
	init = function()
		include( NAB.bin .. "server.lua.txt" )
	end
end

if CLIENT and player() == owner() or SERVER then
	net.sync( "load", nil, init )
end
